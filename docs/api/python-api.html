Python API Reference

Complete Python SDK for controlling autonomous vehicles with Cule OS.

Installation: pip install cule-os

Installation

# Install from PyPI
pip install cule-os

# Or install from source
git clone https://github.com/sanjaysparker27/cule-os.git
cd cule-os
pip install -e .

Connection

UAV.connect(port, baud=115200, **kwargs)

Connect to a vehicle via serial port or network.

Parameters:

port str - Serial port (e.g., '/dev/ttyUSB0') or IP:port for UDP
baud int - Baud rate for serial connection (default: 115200)
wait_ready bool - Wait for vehicle to be ready (default: True)

Connection Examples

# USB connection to flight controller
from cule import UAV

drone = UAV.connect('/dev/ttyUSB0', baud=921600)

# Network connection (UDP)
drone = UAV.connect('udp:192.168.1.10:14550')

# TCP connection to SITL simulator
drone = UAV.connect('tcp:127.0.0.1:5760')

Vehicle Control

Arming and Disarming

drone.arm(check=True)

Arm the vehicle motors.

drone.disarm()

Disarm the vehicle motors.

Flight Modes

drone.mode = mode_name

Set the flight mode. Common modes: 'STABILIZE', 'ALT_HOLD', 'LOITER', 'AUTO', 'RTL', 'LAND', 'GUIDED'

# Set flight mode
drone.mode = 'LOITER'

# Check current mode
print(drone.mode)  # 'LOITER'

Takeoff and Landing

drone.takeoff(altitude, wait=True)

Take off to specified altitude (meters).

drone.land(wait=True)

Land at current position.

drone.rtl()

Return to launch point and land.

# Simple takeoff sequence
drone.arm()
drone.takeoff(altitude=10)  # 10 meters
drone.mode = 'LOITER'

# Return and land
drone.rtl()

Mission Planning

Waypoint Navigation

drone.goto(lat, lon, alt, groundspeed=None)

Navigate to a GPS coordinate.

# Navigate to waypoint
drone.goto(
    lat=37.7749,
    lon=-122.4194,
    alt=50,  # meters
    groundspeed=10  # m/s
)

Mission Upload

drone.upload_mission(waypoints)

Upload a mission to the vehicle.

# Create and upload mission
mission = [
    (37.7749, -122.4194, 30),   # WP 1: Takeoff area
    (37.7759, -122.4184, 50),   # WP 2
    (37.7769, -122.4174, 50),   # WP 3
    (37.7749, -122.4194, 30),   # WP 4: Return
]

drone.upload_mission(mission)
drone.mode = 'AUTO'  # Start mission

Sensor Access

Telemetry Data

drone.telemetry

Access current telemetry data.

# Get telemetry
telem = drone.telemetry
print(f"Position: {telem.lat}, {telem.lon}, {telem.alt}")
print(f"Attitude: roll={telem.roll}, pitch={telem.pitch}, yaw={telem.yaw}")
print(f"Battery: {telem.battery_voltage}V, {telem.battery_percent}%")
print(f"GPS: {telem.gps_sats} sats, fix={telem.gps_fix_type}")

Attitude and Position

drone.attitude

Current roll, pitch, yaw in degrees.

drone.location

Current lat, lon, alt, relative_alt.

Complete Example

from cule import UAV
import time

# Connect to vehicle
print("Connecting...")
drone = UAV.connect('/dev/ttyUSB0')
print("Connected!")

# Arm and takeoff
print("Arming...")
drone.arm()
print("Taking off...")
drone.takeoff(altitude=10)

# Wait for stable hover
time.sleep(5)

# Navigate to waypoint
print("Going to waypoint...")
drone.goto(lat=37.7749, lon=-122.4194, alt=20)

# Wait for arrival
while drone.location.distance_to(37.7749, -122.4194) > 1:
    time.sleep(0.5)

print("At waypoint!")

# Return and land
print("Returning to launch...")
drone.rtl()

# Wait for landing
while drone.telemetry.is_armed:
    time.sleep(1)

print("Mission complete!")

Error Handling

from cule import UAV, CuleError

try:
    drone = UAV.connect('/dev/ttyUSB0')
    drone.arm()
except CuleError.TimeoutError:
    print("Connection timeout - check cable")
except CuleError.ArmError as e:
    print(f"Cannot arm: {e.message}")
finally:
    drone.close()
